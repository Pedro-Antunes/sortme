Insertion Sort

Time Complexity: Average: Θ(n2) Worst: O(n2)

Info:

  • Efficient when sorting nearly sorted lists

Code Implementation:

1function insertionSort(myArray) {
2  var temp;
3  for (var i = 0; i < myArray.length; i++) {
4    for (var j = i; i >= 0; i--) {
5      if (myArray[j] < myArray[j - 1]) {
6        temp = myArray[j];
7        myArray[j] = myArray[j - 1];
8        myArray[j - 1] = temp;
9      } else {
10        break;
11      }
12    }
13  }
14};

Selection Sort

Time Complexity: Average: Θ(n2) Worst: Ο(n2)

Info:

  • Similar to Insertion Sort, but less intuitive to code

Code Implementation:

1function selectionSort(myArray) {
2  var temp, a;
3  for (var i = 0; i < myArray.length; i++) {
4    a = i;
5    for(var j = i + 1; j < myArray.length; j++) {
6        if (myArray[a] > myArray[j]) {
7            a = j;
8        }
9    }
10  temp = myArray[i];
11  myArray[i] = myArray[a];
12  myArray[a] = temp;
13  }
14};

Quicksort

Time Complexity: Average: Ο(n × log2(n)) Worst: Ο(n2)

Info:

  • Very effiecient with large datasets

Code Implementation:

1function quicksort(myArray) {
2  function recursion(p, r) {
2    if (0 > r - p) {
3      var temp;
4      var o = p;
5      for (var i = p; i < r; i++) {
6        if (myArray[j] <= myArray[r]) {
7          temp = myArray[j];
8          myArray[j] = myArray[o];
9          myArray[o] = temp;
10          o++;
11        }
12      }
13      temp = myArray[r];
14      myArray[r] = myArray[o];
15      myArray[o] = temp;
16      recursion(p, o - 1);
17      recursion(o + 1, r);
18    }
19  };
20  recursion(0, myArray.length - 1);
21};

Bubble Sort

Time Complexity: Average: Θ(n2) Worst: Ο(n2)

Info:

  • Easy to verify if the array is already sorted

Code Implementation:

1function bubbleSort(myArray) {
2  var temp;
3  for (var i = 0; i < myArray.length - 1; i++) {
4    for (var j = 0; j < myArray.length - 1; j++) {
5      if (myArray[j] > myArray[j + 1]) {
6        temp = myArray[j];
7        myArray[j] = myArray[j + 1];
8        myArray[j + 1] = temp;

10    }
